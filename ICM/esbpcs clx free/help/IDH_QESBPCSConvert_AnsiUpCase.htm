function AnsiUpCase (const Ch: Char): Char;
{$IFDEF MSWINDOWS}
var
     PC: PChar;
begin
     New (PC);
     try
          PC^ := Ch; // Convert Char to PChar
          Result := CharUpper (PC)^; // Call Windows function
     finally
          Dispose (PC);
     end;
end;
{$ENDIF}
{$IFDEF LINUX}
begin
     Result := Char (towupper (UCS4Char (Ch)));
End;
