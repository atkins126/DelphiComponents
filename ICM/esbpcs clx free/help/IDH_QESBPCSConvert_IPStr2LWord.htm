function IPStr2LWord (const IPAddr: string): LongWord;
var
     I, P: Integer;
     S: string;
     X: Longword;
begin
     Result := 0;
     S := IPAddr;
     for I := 1 to 4 do
     begin
          P := ESBPosCh ('.', S);
          if P > 0 then
          begin
               X := Str2LWord (LeftStr (S, P - 1));
               S := RightAfterStr (S, P);
          end
          else
          begin
               X := Str2LWord (S);
               S := '';
          end;

          if (X > 255) then
          begin
               Result := 0;
               if ESBRaiseIPError then
                    raise EMathError.Create (rsIPAddrComp)
          end
          else
               Result := Result * Int64 (256) + X;
     end;
End;
