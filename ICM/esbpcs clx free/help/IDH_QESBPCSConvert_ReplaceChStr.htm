function ReplaceChStr (const S: string; const OldCh, NewCh: Char): string;
var
     I: LongWord;
begin
     Result := S;
     if OldCh = NewCh then
          Exit;
     for I := 1 to Length (S) do
          if S [I] = OldCh then
               Result [I] := NewCh;
End;
