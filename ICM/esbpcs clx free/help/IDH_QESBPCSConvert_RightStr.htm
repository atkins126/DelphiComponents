function RightStr (const S: string; const N: LongWord): string;
var
     M: LongWord;
begin
     M := Int64 (Length (S)) - N + 1; // Calculate the starting point
     if M < 1 then // If longer then string, return string
          M := 1;
     Result := Copy (S, M, N); // Return last N characters
End;
