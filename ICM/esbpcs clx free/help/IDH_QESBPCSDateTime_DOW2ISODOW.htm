function DOW2ISODOW (const DOW: Byte): Byte;
begin
     if (DOW < 1) or (DOW > 7) then
          raise EConvertError.Create (rsInvalidDOW);

     Result := Dow - 1;
     if Result = 0 then
          Result := 7;
End;
