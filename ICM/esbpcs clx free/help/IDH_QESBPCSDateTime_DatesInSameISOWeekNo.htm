function DatesInSameISOWeekNo (const DT1, DT2: TDateTime): Boolean;
var
     W1, W2: Integer;
     Y1, Y2: Word;
begin
     Date2ISOWeekNo (DT1, W1, Y1);
     Date2ISOWeekNo (DT2, W2, Y2);
     Result := (W1 = W2) and (Y1 = Y2);
End;
