function GetFirstMondayOfYear (const Year: Word): TDateTime;
var
     StartYear: TDateTime;
begin
     StartYear := GetFirstDayOfYear (Year);
     if ISODayOfWeek (StartYear) = 1 then
          Result := StartYear
     else
          Result := StartOfISOWeek (StartYear) + 7;
End;
