function GetLastDayofQuarter (const DT: TDateTime): TDateTime;
var
     D, M, Y: Integer;
begin
     OptDecodeDateI (DT, Y, M, D);
     case M of
          1..3:
               begin
                    M := 3;
                    D := 31;
               end;
          4..6:
               begin
                    M := 6;
                    D := 30;
               end;
          7..9:
               begin
                    M := 9;
                    D := 30;
               end;
          10..12:
               begin
                    M := 12;
                    D := 31;
               end;
     end;
     Result := OptEncodeDateI (Y, M, D) + Frac (DT);
End;
