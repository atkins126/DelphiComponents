function IsValidLongMonth (const Month: string): Boolean;
var
     S: string;
     I: Integer;
begin
     Result := False;
     S := AnsiUpperCase (Month);
     for I := 1 to 12 do
          if S = AnsiUpperCase (LongMonthNames [I]) then
          begin
               Result := True;
               Break
          end;
End;
