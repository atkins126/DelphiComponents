procedure Deg2DMS (const Deg: Extended; out Degs, Mins, Secs: Extended;
     out Sign: ShortInt);
var
     X: Extended;
begin
     Sign := ESBSign (Deg);
     Degs := Int (Deg);
     X := Frac (Abs (Deg)) * 60;
     Mins := Int (X);
     Secs := Frac (X) * 60;
End;
