procedure ValidationMsg (const Field, Msg: string);
var
     Hold: TCursor;
begin
     Hold := Screen.Cursor;
     Screen.Cursor := crDefault;
     try
          MessageDlg (rsField + ': ' + Field + #13 + RsReason + ': ' + Msg,
               mtError, [mbOK], 0);
     finally
          Screen.Cursor := Hold;
     end;
End;
