function Bits2LStr (const BitList: TESBLongBitList): String32;
var
     I: Integer;
begin
     SetLength (Result, 32);
     for I := 0 to 31 do
          if BitIsSet (BitList, I) then
               Result [I + 1] := '1'
          else
               Result [I + 1] := '0';
End;
